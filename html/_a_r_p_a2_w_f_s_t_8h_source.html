40  /*
41  Transform a statistical language model in ARPA format
42  to an equivalent Weighted Finite-State Acceptor.
43  This implementation adopts the Google format for the output
44  WFSA. This differs from previous implementations in several ways:
45 
46  Start-state and <s> arcs:
47  * There are no explicit sentence-begin (<s>) arcs
48  * There is a single <s> start-state.
49 
50  Final-state and </s> arcs:
51  * There are no explicit sentence-end (</s>) arcs
52  * There is no explicit </s> state
53  * NGrams ending in </s> are designated as final
54  states, and any probability is assigned
55  to the final weight of said state.
56  */

106  void arpa_to_wfst () {
107  /*
108  Convert an ARPA format Statistical Language Model to WFST format suitable
109  for use with phonetisaurus-g2p. In this implementation we adopt the Google
110  convention where the the sentence-begin (<s>) and sentence-end (</s>) tokens
111  are represented *implicitly*, without arcs, and the model has multiple final
112  states.
113 
114  This simplifies downstream processing for pronunciation generation, and has
115  the added desirable side-effect of achieving a significant reduction in the
116  number of transitions in the resulting WFST model.
117 
118  The model is expected to be in the following standardized format:
119 
120  \data\
121  ngram 1=M
122  ngram 2=M
123  ...
124  ngram N=M
125 
126  \1-grams:
127  p(w) w bow(w)
128  ...
129  \2-grams:
130  p(v,w) v w bow(v,w)
131  ...
132  \3-grams:
133  p(u,v,w) u v w
134 
135  \end\
136 
137  where M refers to the number of unique NGrams for this order,
138  and N refers to the maximum NGram order of the model.
139  Similarly, p(w) refers to the probability of NGram 'w', and
140  bow(w) refers to the back-off weight for NGram 'w'. The highest
141  order of the model does not have back-off weights. Back-off
142  weights equal to 0.0 in log-base 10 may be omitted to save space,
143  and NGrams ending in sentence-end (</s>) naturally do not have
144  back-off weights.
145 
146  The NGram columns are separated by a single tab (\t).
147 
148  */
149 
150  if (arpa_lm_fp.is_open ()) {
151  while (arpa_lm_fp.good ()) {
152  getline (arpa_lm_fp, line);
153  if (current_order > 0 && line.compare ("") != 0 \
154  && line.compare (0,1,"\\") != 0) {
155  //Split the input using '\s+' as a delimiter
156  vector<string> ngram;
157  istringstream iss (line);
158  copy (istream_iterator<string>(iss),
159  istream_iterator<string>(),
160  back_inserter<vector<string> >(ngram));
161  double prob = atof (ngram.front ().c_str ());
162  ngram.erase (ngram.begin ());
163  double bow = 0.0;
164  if (ngram.size () > current_order) {
165  bow = atof (ngram.back ().c_str ());
166  ngram.pop_back ();
167  }
168  //We have a unigram model
169  if (max_order == 1) {
170  //Assume unigram ARPA model has a <s>
171  // sentence-begin line. Is this true?
172  if (ngram.front ().compare (sb) == 0)
173  continue;
174  else if (ngram.front ().compare (se) == 0) {
175  _make_final (sb, log10_2tropical (prob));
176  }else
177  _make_arc (sb, sb, ngram.at (0), prob);
178  //We have a higher order model
179  }else if (current_order == 1) {
180  if (ngram.front ().compare (sb) == 0) {
181  _make_arc (sb, eps, eps, bow);
182  }else if (ngram.back ().compare (se) == 0) {
183  _make_final (eps, prob);
184  }else{
185  _make_arc (eps, ngram.front (), ngram.front (), prob);
186  _make_arc (ngram.front (), eps, eps, bow);
187  }
188  }else if (current_order < max_order) {
189  string isym = ngram.back ();
190  string s_st = _join (ngram.begin (), ngram.end () - 1);
191  if (isym.compare (se) == 0) {
192  _make_final (s_st, prob);
193  }else{
194  string e_st = _join (ngram.begin (), ngram.end ());
195  string b_st = _join (ngram.begin () + 1, ngram.end ());
196  _make_arc (s_st, e_st, isym, prob);
197  _make_arc (e_st, b_st, eps, bow);
198  }
199  }else if (current_order == max_order) {
200  string isym = ngram.back ();
201  string s_st = _join (ngram.begin (), ngram.end () - 1);
202  if (isym.compare (se) == 0) {
203  _make_final (s_st, prob);
204  }else{
205  string e_st = _join (ngram.begin() + 1, ngram.end ());
206  _make_arc (s_st, e_st, isym, prob);
207  }
208  }
209  }else if (line.size() > 4 && line.compare (0, 5, "ngram") == 0) {
210  for (size_t i = 0; i < line.size (); i++)
211  if (line.compare (i, 1, "=") == 0)
212  line.at (i) = ' ';
213  vector<string> parts;
214  istringstream iss (line);
215  copy (istream_iterator<string>(iss),
216  istream_iterator<string>(),
217  back_inserter<vector<string> >(parts));
218  //Make sure there is at least one n-gram for max order!
219  if (atoi (parts [2].c_str()) > 0)
220  max_order = (size_t)atoi (parts [1].c_str ()) > max_order \
221  ? atoi (parts [1].c_str ()) : max_order;
222  //cerr << "MaxOrder: " << max_order << endl;
223  //max_order = (size_t)atoi(&line[6])>max_order ? atoi(&line[6]) : max_order;
224  }else if (line.compare ("\\data\\") == 0) {
225  continue;
226  }else if (line.compare ("\\end\\") == 0) {
227  break;
228  }else if (line.size() > 0 && line.compare (0, 1, "\\") == 0) {
229  line.replace (0, 1, "");
230  if (line.compare ( 1, 1, "-" ) == 0)
231  line.replace (1, 7, "");
232  else //Will work up to N=99.
233  line.replace (2, 7, "");
234  current_order = atoi (&line [0]);
235  }
236  }
237  arpa_lm_fp.close();
238 
239  _patch_labels (isyms, true);
240  _patch_labels (osyms, false);
241  arpafst.SetInputSymbols (isyms);
242  arpafst.SetOutputSymbols (osyms);
243  }else{
244  cout << "Unable to open file: " << arpa_lm_file << endl;
245  }
246  }

249  double log10_2tropical (double val) {
250  /*
251  Convert an ARPA-standard log10(val) value to the
252  tropical/log semiring, which is -logE(val).
253  Make sure that the result is a valid number.
254 
255  The decoder will work even if the values are BadNumbers
256  or Infinity, but fstinfo will treat the result as malformed.
257  */
258 
259  val = log(10.0) * val * -1.0;
260 
261  if( !(val <= DBL_MAX && val >= -DBL_MAX) )
262  val = 999.0;
263 
264  if( val != val )
265  val = 999.0;
266 
267  return val;
268  }

270  void _make_arc (string istate, string ostate, string isym, double weight) {
271  //Build up an arc for the WFST. Weights default to the Log semiring.
272  int is_id = ssyms->Find (istate);
273  int os_id = ssyms->Find (ostate);
274  if (is_id == -1) {
275  is_id = arpafst.AddState();
276  ssyms->AddSymbol (istate, is_id);
277  }
278  if (os_id == -1) {
279  os_id = arpafst.AddState ();
280  ssyms->AddSymbol (ostate, os_id);
281  }
282  weight = log10_2tropical (weight);
283 
284  vector<string> io = tokenize_utf8_string (&isym, &split);
285  if (io.size () == 2) {
286  /*
287  Keep everything as it is: helpful for rescoring and reranking.
288  if( io[0]==skip )
289  io[0] = eps;
290  if( io[1]==skip )
291  io[1] = eps;
292  */
293  arpafst.AddArc (is_id, StdArc (isyms->AddSymbol (io [0]),
294  osyms->AddSymbol (io [1]),
295  weight, os_id));
296  }else{
297  arpafst.AddArc (is_id, StdArc (isyms->AddSymbol (isym),
298  osyms->AddSymbol (isym),
299  weight, os_id));
300  }
301 
302  return;
303  }

334  void _patch_labels (SymbolTable* syms, bool input) {
335  /*
336  Patch all labels. In some edge cases it is possible
337  to end up grapheme subsequences: e.g. 'QU' where one or
338  both tokens is only mapped to the multi-subsequence. In thise
339  case the independent 'Q' and/or 'U' token will never appear
340  in isolation.
341  This bit resolves this by:
342 
343  a.) finding and adding these missing subsequence tokens
344  b.) adding backoff loops to the LM
345 
346  */
347  for (unsigned int i = 3; i < syms->NumSymbols(); i++) {
348  string sym = syms->Find (i);
349  vector<string> parts = tokenize_utf8_string (&sym, &tie);
350  if (parts.size() > 1) {
351  for (unsigned int j = 0; j < parts.size (); j++) {
352  if (syms->Find (parts [j]) == -1) {
353  //Add the missing symbol
354  int k = syms->AddSymbol (parts [j]);
355  //Add a backoff loop mapped to the 'skip'
356  if (input == true)
357  arpafst.AddArc (1, StdArc (k, 2, 99, 1));
358  else
359  arpafst.AddArc (1, StdArc (2, k, 99, 1));
360  }
361  }
362  }
363  }
